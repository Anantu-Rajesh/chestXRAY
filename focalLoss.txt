# Define Focal Loss for imbalanced classification
def focal_loss(gamma=2.0, alpha=0.25):
    def focal_loss_fixed(y_true, y_pred):
        epsilon = tf.keras.backend.epsilon()
        y_pred = tf.clip_by_value(y_pred, epsilon, 1.0 - epsilon)
        
        # Calculate focal loss
        y_true = tf.cast(y_true, tf.float32)
        cross_entropy = -y_true * tf.math.log(y_pred)
        
        # Apply focal term: (1 - pt)^gamma
        focal_weight = tf.pow(1.0 - y_pred, gamma) * y_true + \
                       tf.pow(y_pred, gamma) * (1.0 - y_true)
        
        # Apply alpha balancing
        alpha_weight = alpha * y_true + (1 - alpha) * (1.0 - y_true)
        
        loss = alpha_weight * focal_weight * cross_entropy
        return tf.reduce_mean(loss)
    
    return focal_loss_fixed
